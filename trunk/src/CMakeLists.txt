FIND_PACKAGE( G15Tools REQUIRED )
FIND_PACKAGE( LibDaemon REQUIRED )
FIND_PACKAGE( GLIB2 )

FIND_PACKAGE( DBUS REQUIRED )
INCLUDE_DIRECTORIES(${DBUS_INCLUDES}

					${DBUS_GLIB_INCLUDE_DIR}
					${GLIB2_INCLUDE_DIR}

					${DBUS_GLIB_LIBRARY}
					${DBUS_LIBRARY}
					${GLIB2_LIBRARIES}
					${LIBDAEMON_LIBRARY}
					${G15TOOLS_LIBRARY} )
